  <Step num="1" title="Acesse o Editor de Avatar do Roblox">

  <Step num="2" title="Navegue por Roupas e Acessórios">

  <Step num="3" title="Equipe Seus Itens Escolhidos">

  </Step>
  <Step num="4" title="Salve as Mudanças do Seu Avatar">

  <Step num="5" title="Reentre em Piggy Intercity">
